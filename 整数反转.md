  def reverse(self, x: int) -> int:

​    x2 = abs(x)



​    while True:

​      \# 注意先求anum,再对x2的值进行更新

​      res = res * 10 + anum

​        break



​    \# 边界条件

​      return res

​    if x < 0 and -res >= -pow(2,31):

​      return -res

​    return 0
